fnGetUser
(userId as nullable number) =>

let
    Token = "C869OUgvkJ5on8fSlYX7qerEKGkVxw0HA80ge",
    Result =
        if userId = null then
            [
                UserId = null,
                UserName = null,
                UserEmail = null
            ]
        else

        let
            Response =
                try
                    Json.Document(
                        Web.Contents(
                            "https://api.smartsheet.com",
                            [
                                RelativePath = "2.0/users/" & Number.ToText(userId),
                                Headers = [
                                    Authorization = "Bearer " & Token,
                                    Accept = "application/json"
                                ]
                            ]
                        )
                    )
                otherwise
                    null
        in
            if Response = null then
                [
                    UserId = userId,
                    UserName = null,
                    UserEmail = null
                ]
            else
                [
                    UserId = userId,
                    UserName = try Response[name] otherwise null,
                    UserEmail = try Response[email] otherwise null
                ]

in
    Result
